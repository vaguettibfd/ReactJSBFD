==== Login.css ====

.login-container {
  display: flex;               /* usa Flexbox para organizar os elementos filhos */
  justify-content: center;     /* centraliza horizontalmente */
  align-items: center;         /* centraliza verticalmente */
  height: 100vh;               /* ocupa 100% da altura da tela (viewport) */
  background-color: #f5f5f5;   /* cor de fundo cinza claro */
}

.login-form {
  background: white;                           /* fundo branco */
  padding: 2rem;                               /* espaço interno (respiro) */
  border-radius: 8px;                          /* cantos arredondados */
  box-shadow: 0px 4px 10px rgba(0, 0, 0, 0.1); /* sombra leve para destacar */
  width: 320px;                                /* largura fixa do formulário */
}

.form-group {
  margin-bottom: 1rem;        /* espaço inferior entre grupos de campos */
  display: flex;              /* ativa Flexbox */
  flex-direction: column;     /* empilha os elementos filhos (label e input) na vertical */
}

==== LabelTitle.css ====

.labelTitle {

  margin-bottom: 0.3rem; /* espaço inferior pequeno para separar do próximo elemento */

}

==== Label.css ====

.label {
  display: block;        /* ocupa toda a largura → força o label a ficar acima do input */
  font-size: 1.2rem;     /* tamanho de fonte médio (um pouco maior que o padrão) */
  margin-bottom: 0.3rem; /* espaço inferior pequeno para separar do input */
  color: #333;           /* cinza escuro, boa leitura sem ser preto puro */
}

==== InputText.css ====

input[type="text"] {
  width: 100%;              /* ocupa toda a largura disponível do container */
  padding: 0.6rem;          /* espaço interno (respiro) */
  font-size: 0.95rem;       /* fonte um pouco menor que 1rem (~15px) */
  border: 1px solid #ccc;   /* borda cinza clara */
  border-radius: 4px;       /* cantos arredondados */
  outline: none;            /* remove o contorno azul padrão do navegador */
  transition: border-color 0.3s ease; /* animação suave ao trocar cor da borda */
  box-sizing: border-box;   /* inclui padding e borda no cálculo da largura */
  margin-bottom: 0.3rem;    /* espaço inferior para separar do próximo elemento */
}

input[type="text"]:focus {
  border-color: #0066cc;               /* azul quando selecionado */
  box-shadow: 0 0 3px rgba(0, 102, 204, 0.4); /* Sombra da caixa - brilho azul suave */
}

p {
  font-size: 0.85rem;     /* texto menor que o normal */
  color: #cc0000;         /* vermelho para chamar atenção */
  margin: 0.2rem 0 0;     /* espaço pequeno acima, sem espaço extra embaixo */
}

==== InputPassword.css ====

input[type="password"] {
  width: 100%;              /* ocupa toda a largura disponível */
  padding: 0.6rem;          /* espaço interno para o texto não grudar nas bordas */
  font-size: 0.95rem;       /* tamanho de fonte ligeiramente menor que 1rem (~15px) */
  border: 1px solid #ccc;   /* borda cinza clara padrão */
  border-radius: 4px;       /* cantos arredondados */
  outline: none;            /* remove o contorno azul padrão dos navegadores */
  transition: border-color 0.3s ease; /* transição suave ao mudar a cor da borda */
  box-sizing: border-box;   /* inclui padding e borda no cálculo da largura */
  margin-bottom: 0.3rem;    /* espaço abaixo do input */
}

input[type="password"]:focus {
  border-color: #28a745; /* verde */
  box-shadow: 0 0 3px rgba(40, 167, 69, 0.4); /* Sombra da caixa - verde*/
}

p {
  font-size: 0.85rem;   /* fonte menor que a padrão */
  color: #555;          /* cinza médio para não competir com o input */
  margin: 0.2rem 0 0;   /* espaço pequeno acima, sem margem extra embaixo */
  font-style: italic;   /* texto em itálico para diferenciar */
}

==== Button.css ====

.btn {
